function printEmojis() {
    const nodes = document.querySelectorAll('.echars');

    const dict = {};

    nodes
        .forEach((entry) => {
            const text = entry.innerHTML;

            if(!dict[text.length]) {
                dict[text.length] = [];
            }

            dict[text.length].push(text);
        });

    console.log(dict);
}

printEmojis();
